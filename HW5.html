# Question 5  
# create a heatmap for the daily_sea_temps.csv dataset for the past 20 years (2004-2024)   
# where the rows are year and the columns are months  
# and the value is the monthly median sst value  
# use the seismic colormap and annotate each square with the value  
# rounded to 1 decimal point  

filtered_df = daily_sea_temps_df[(daily_sea_temps_df['date'].dt.year >= 2004) & (daily_sea_temps_df['date'].dt.year <= 2024)].copy()

# Adding year and month columns
filtered_df['year'] = filtered_df['date'].dt.year
filtered_df['month'] = filtered_df['date'].dt.month

# Calculating the monthly median SST for each year and month
monthly_median = filtered_df.groupby(['year', 'month'])['sst'].median().unstack()

# Creating the heatmap
plt.figure(figsize=(12, 8))
sns.heatmap(monthly_median, annot=True, fmt=".1f", cmap='seismic', linewidths=.5, cbar_kws={'label': 'Median SST (°C)'})

# Adding labels and title
plt.xlabel('Month')
plt.ylabel('Year')
plt.title('Heatmap of Monthly Median SST (2004-2024)')
plt.show()
